%{
#include <stdio.h>
#include <stdlib.h>

int yylex(void);
void yyerror(const char *s);
%}

%token NUMBER
%left '+' '-'
%left '*' '/'
%left UMINUS

%%
input:
      /* empty */
    | input line
    ;

line:
      '\n'
    | expr '\n'        { printf("Result = %d\n", $1); printf("\nEnter expression: "); }
    | error '\n'       { printf("Error: Invalid expression. Try again.\n"); yyerrok; printf("\nEnter expression: "); }
    ;

expr:
      NUMBER
    | expr '+' expr    { $$ = $1 + $3; }
    | expr '-' expr    { $$ = $1 - $3; }
    | expr '*' expr    { $$ = $1 * $3; }
    | expr '/' expr    { if($3==0) { printf("Error: Division by zero.\n"); $$=0; } else $$=$1/$3; }
    | '(' expr ')'     { $$ = $2; }
    | '-' expr %prec UMINUS { $$ = -$2; }
    ;
%%

void yyerror(const char *s) {
    fprintf(stderr, "Syntax error: %s\n", s);
}

int main() {
    printf("===== Desk Calculator =====\n");
    printf("Type arithmetic expressions and press Enter.\n");
    printf("Supports +, -, *, /, and parentheses.\n");
    printf("Ctrl+C to exit.\n\n");
    printf("Enter expression: ");

    yyparse();
    return 0;
}
